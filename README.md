### ListSelectListItem.GetLista

public class Todo
{

    [Key]
    
    public int Id { get; set; }
    
    [LabelForDropDown]
    
    public string Name { get; set; 
    
    public string Code { get; set; }
    
}


var list = new List<Todo>();

list.Add(new Todo() { Id = 1 , Name = "Todo 1" , Code = "T1" });
